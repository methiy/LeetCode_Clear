class Solution {
public:
    
    vector<int> findDiagonalOrder(vector<vector<int>>& mat) {
        vector<int> v;
        
        int len=max(mat.size(),mat[0].size());
        
        int x=0,y=0;
        for(int i=1;i<=len;i++){
            for(int j=1;j<=i;j++){
                if(j!=1){
                    if(i&1) x--,y++;
                    else x++,y--;
                }
                if(x>=0&&x<mat.size()&&y>=0&&y<mat[0].size())
                    v.push_back(mat[x][y]);
            }
            if(i&1){
                y++;
            }else{
                x++;
            }
        }
        if(len&1)   x=1,y=len-1;
        else y=1,x=len-1;
        for(int i=len-1;i>=0;i--){
            for(int j=1;j<=i;j++){
                if(j!=1){
                    if(i&1) x--,y++;
                    else x++,y--;
                }
                if(x>=0&&x<mat.size()&&y>=0&&y<mat[0].size())
                    v.push_back(mat[x][y]);
            }
            if(i&1){
                x++;
            }else{
                y++;
            }
        }
        
        return v;
    }
};
